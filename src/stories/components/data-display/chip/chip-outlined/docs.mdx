### Import
```js
import Chip from 'qasir-ui/data-display/Chip';
```

### Usage

```html
      <Chip variant="outlined" label="Basic" />
```
```html 
      <Chip variant="outlined" label="Disabled" disabled />
```
### Function HanldeClick for example Usage
```html
  const handleClick = () => {
    alert('You clicked the Chip.');
  };

  const handleDelete = () => {
    alert('You clicked the delete icon.');
  };

```
