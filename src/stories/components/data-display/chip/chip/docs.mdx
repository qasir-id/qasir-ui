# Chip

<Subtitle>
  Chips are compact elements that represent an input, attribute, or action.
</Subtitle>
<Preview withSource="none">
  <Story id="components-data-display-chip-example--chip" />

### Import
```js
import Chip from 'qasir-ui/data-display/Chip';
```

### Usage

```jsx
      <Chip label="Basic" />
```
```jsx 
      <Chip label="Disabled" disabled />
```
### Event Handler
```js
  const handleClick = () => {
    alert('You clicked the Chip.');
  };

  const handleDelete = () => {
    alert('You clicked the delete icon.');
  };

```

```jsx
      <Chip avatar={<Avatar>M</Avatar>} label="Clickable" onClick={handleClick} />
```
```jsx  
<Chip
  avatar={<Avatar alt="Natacha" src="https://static.toiimg.com/photo/msid-67586673/67586673.jpg?3918697" />}
  label="Deletable"
  onDelete={handleDelete}
/>
```
```jsx
<Chip
  icon={<FaceIcon />}
  label="Clickable deletable"
  onClick={handleClick}
  onDelete={handleDelete}
/>
```
```jsx
<Chip
  label="Custom delete icon"
  onClick={handleClick}
  onDelete={handleDelete}
  deleteIcon={<DoneIcon />}
/>
```
```jsx
      <Chip label="Clickable Link" component="a" href="#chip" clickable />
```
```jsx 
<Chip
  avatar={<Avatar>M</Avatar>}
```
