```js
import TextField from 'qasir-ui/inputs/TextField';
```

```js
const currencies = [
  {
    value: 'USD',
    label: '$',
  },
  {
    value: 'EUR',
    label: '€',
  },
  {
    value: 'BTC',
    label: '฿',
  },
  {
    value: 'JPY',
    label: '¥',
  },
];

const funcExample = () => {
  const [currency, setCurrency] = React.useState('EUR');

  const handleChange = (event) => {
    setCurrency(event.target.value);
  };
  return (
    <>
      ...Components Select
    </>
  );
};
```

```html
<TextField
  select
  label="Select"
  value={currency}
  onChange={handleChange}
  helperText="Please select your currency"
  variant="outlined"
  >
    {currencies.map((option) => (
      <MenuItem key={option.value} value={option.value}>
        {option.label}
      </MenuItem>
    ))}
</TextField>
```

```html
<TextField
  select
  label="Native select"
  value={currency}
  SelectProps={{ native: true }}
  onChange={handleChange}
  helperText="Please select your currency"
  variant="outlined"
  >
    {currencies.map((option) => (
      <MenuItem key={option.value} value={option.value}>
        {option.label}
      </MenuItem>
    ))}
</TextField>
```
