```js
import TextField from 'qasir-ui/inputs/TextField';
```

```html
<TextField required label="Required*" defaultValue="Hello World" variant="outlined" />
```

```html
<TextField disabled label="Disabled" defaultValue="Hello World" variant="outlined" />
```

```html
<TextField label="Password" type="password" autoComplete="current-password" variant="outlined" />
```

```html
<TextField label="Read Only" defaultValue="Hello World" InputProps={{ readOnly: true }} variant="outlined" />
```

```html
<TextField label="Number" type="number" InputLabelProps={{ shrink: true }} variant="outlined" />
```

```html
<TextField label="Search field" type="search" variant="outlined" />
```

```html
<TextField label="Helper text" defaultValue="Hello World" helperText="Some important text" variant="outlined" />
```
