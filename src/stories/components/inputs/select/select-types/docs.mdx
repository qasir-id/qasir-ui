import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import FormControl from '@material-ui/core/FormControl';
import MenuItem from '@material-ui/core/MenuItem';
import Select from 'components/inputs/Select';
import InputLabel from 'components/inputs/InputLabel';

<Title>Type Selects</Title>
<Subtitle>Menus are positioned over their emitting elements such that the currently selected menu item appears on top of the emitting element.</Subtitle>

### Import

```js
import FormControl from '@material-ui/core/FormControl';
import MenuItem from '@material-ui/core/MenuItem';

import Select from 'qasir-ui/inputs/Select';
import InputLabel from 'qasir-ui/inputs/InputLabel';
```

### Usage

<FormControl variant="outlined" style={{ minWidth: 150, marginTop: 20 }}>
  <InputLabel>Simple Select</InputLabel>
  <Select
    label="Simple Select"
    inputProps={{
      name: 'simple-select',
    }}
  >
    <MenuItem value="">
      <em>None</em>
    </MenuItem>
    <MenuItem value={10}>Ten</MenuItem>
    <MenuItem value={20}>Twenty</MenuItem>
    <MenuItem value={30}>Thirty</MenuItem>
  </Select>
</FormControl>
