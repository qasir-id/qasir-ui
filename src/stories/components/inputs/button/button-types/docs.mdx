import {ButtonPrimary, ButtonSecondary, ButtonDisable, ButtonOutline, ButtonText} from './index';

<Title>Type Buttons</Title>
<Subtitle>There are text, contained, and toggle buttons. Text buttons are typically used for less important actions.</Subtitle>

### Import

```jsx
import Button from 'qasir-ui/inputs/Button';
```

### Usage

<ButtonPrimary style={{marginTop: 20}} />

```jsx
<Button variant="contained" color="primary">Primary</Button>
```

<ButtonSecondary />

```jsx
<Button variant="contained" color="secondary">Secondary</Button>
```

<ButtonDisable />

```jsx
<Button variant="contained" disabled>Disable</Button>
```

<ButtonOutline />

```jsx
<Button variant="outlined">Outline</Button>
```

<ButtonText />

```jsx
<Button variant="text">Text</Button>
```
