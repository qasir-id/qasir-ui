you can see below, size and type of Checkbox

### Import

```js
import Checkbox from 'qasir-ui/inputs/Checkbox';
import FormControlLabel from 'qasir-ui/inputs/FormControlLabel';
```

### Usage

<PrimaryCheckbox />

```jsx
const PrimaryCheckbox = () => {
    const [checked, setChecked] = React.useState(false);
    const handleChange = (event) => setChecked(event.target.checked);

    return (
        <FormControlLabel
        control={
            <Checkbox
            checked={checked}
            onChange={handleChange}
            color="primary"
            inputProps={{ 'aria-label': 'primary checkbox' }}
            />
        }
        label="Primary"
        />
    );
};
```

<SecondaryCheckbox />

```jsx
<FormControlLabel
    control={
        <Checkbox defaultChecked color="secondary" inputProps={{ 'aria-label': 'secondary checkbox' }} />
    }
    label="Secondary"
/>
```

<UncontrollableCheckbox />

```jsx
<FormControlLabel
    control={<Checkbox inputProps={{ 'aria-label': 'uncontrolled-checkbox' }} />}
    label="Uncontrolled"
/>
```

<DisabledCheckbox />

```jsx
<FormControlLabel
    control={<Checkbox disabled inputProps={{ 'aria-label': 'disabled checkbox' }} />}
    label="Disabled"
/>
```

<DisabledCheckedCheckbox />

```jsx
 <FormControlLabel
    control={<Checkbox disabled checked inputProps={{ 'aria-label': 'disabled checked checkbox' }} />}
    label="Disabled checked"
/>
```

<IndeterminateCheckbox />

```jsx
<FormControlLabel
    control={
        <Checkbox defaultChecked indeterminate inputProps={{ 'aria-label': 'indeterminate checkbox' }} />
    }
    label="Indeterminate"
/>
```

<CustomIconCheckbox />

```jsx
import Favorite from '@material-ui/icons/Favorite';
import FavoriteBorder from '@material-ui/icons/FavoriteBorder';

<FormControlLabel
    control={
        <Checkbox
          icon={<FavoriteBorder />}
          checkedIcon={<Favorite />}
          inputProps={{ 'aria-label': 'custom icon' }}
        />
    }
    label="Custom icon"
/>
```
