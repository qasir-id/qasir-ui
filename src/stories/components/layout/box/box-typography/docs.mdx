### Import

```jsx

```

<h3 style={{marginBottom: 20}}>Usage</h3>

<BoxTypographyTextAlignment />

        odio operae, eu vulputate felis rhoncus.
      </Box>
      <Box textAlign="left" m={1}>
        Left aligned text.
      </Box>
      <Box textAlign="center" m={1}>
        Center aligned text.
      </Box>
      <Box textAlign="right" m={1}>
        Right aligned text.
      </Box>
  `}
/>
