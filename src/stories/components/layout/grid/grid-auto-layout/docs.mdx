<Title>Auto Layout</Title>
<Subtitle>The Auto-layout makes the items equitably share the available space. That also means you can set the width of one item and the others will automatically resize around it.</Subtitle>

### Import

```jsx
// Vendors
import React from 'react';

// Material UI
import { makeStyles } from '@material-ui/core/styles';

// Qasir UI
import Box from 'qasir-ui/layout/Box';
import Grid from 'qasir-ui/layout/Grid';
```

<h3 style={{marginBottom: 25}}>Usage</h3>

<GridFluidBasic />

```jsx
const useStyles = makeStyles((theme) => ({
  root: {
    flexGrow: 1,
  },
}));

const defaultProps = {
  bgcolor: '#cfe8fc',
  p: 2,
  textAlign: 'center',
};

const GridAutoLayout = () => {
  const classes = useStyles();

  return (
    <div className={classes.root}>
      <Grid container spacing={3}>
        <Grid item xs>
          <Box {...defaultProps}>xs</Box>
        </Grid>
        <Grid item xs>
          <Box {...defaultProps}>xs</Box>
        </Grid>
        <Grid item xs>
          <Box {...defaultProps}>xs</Box>
        </Grid>
      </Grid>
      <Grid container spacing={3}>
        <Grid item xs>
          <Box {...defaultProps}>xs</Box>
        </Grid>
        <Grid item xs={6}>
          <Box {...defaultProps}>xs=6</Box>
        </Grid>
        <Grid item xs>
          <Box {...defaultProps}>xs</Box>
        </Grid>
      </Grid>
    </div>
  );
};

export default GridAutoLayout;
```
