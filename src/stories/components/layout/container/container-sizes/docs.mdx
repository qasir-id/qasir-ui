import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import {ContainerSizeXL, ContainerSizeLG, ContainerSizeMD, ContainerSizeSM, ContainerSizeXS} from './index';

<Title>Container Sizes</Title>
<Subtitle>Determine the max-width of the container. The container width grows with the size of the screen. Set to false to disable maxWidth.</Subtitle>

### Import

```jsx
import Container from 'qasir-ui/inputs/Container';
```

### Usage

<ContainerSizeXL />

```jsx
<Container maxWidth="xl">...</Container>
```

<ContainerSizeLG />

```jsx
<Container maxWidth="lg">...</Container>
```

<ContainerSizeMD />

```jsx
<Container maxWidth="md">...</Container>
```

<ContainerSizeSM />

```jsx
<Container maxWidth="sm">...</Container>
```

<ContainerSizeXS />

```jsx
<Container maxWidth="xs">...</Container>
```
